\begin{theorem}
	Each vertex of $\P$ lies within an $r$-ball centered at the corresponding \\
	vertex of $P$, where
	$r = E^2 \cdot  L \cdot 2 \sin ( \mathcal D \gamma / 2 ) + E \mu$.
\end{theorem} \vspace{-1.2mm}
